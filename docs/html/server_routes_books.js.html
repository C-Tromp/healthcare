/*jslint node: true */

var express = require('express');
var router = express.Router();

"use strict";


/**  book routes
 * ---------------
 * Create a variable "controller" that holds the controller object.
 * We map the URL to a method in the created variable "controller".
 * In this example is a mapping for every CRUD action.
 *
 *     https://github.com/RestDoc/node-docrouter
 *
 * | CRUD operation | http verb | Url                     | Route parameter| Body parameter(s) | Controller |
 * | -------------- | --------- | ----------------------- | -------------- | ----------------- | ---------  |
 * | Create         | POST      | http://localhost:3000/  |                | {title: "", author:"", description: ""} | create |
 * | Get (all)      | GET       | http://localhost:3000/  |                |  | list |
 * | Get (1)        | GET       | http://localhost:3000/  | {id}           |  | detail |
 * | Update         | PUT       | http://localhost:3000/  | {id}           | {title: "", author:"", description: ""} | updateOne |
 * | Delete         | DELETE    | http://localhost:3000/  | {id}           |  | delete |
 *
 *
 * @module Router/Book
 */
var controller = require('../app/controllers/books.js');


/**
 * CREATE route for books
 */
router
    .post('/books', controller.create);

/**
 * GET routes
 * @function
 */
router
    .get('/books', controller.list)
    .get('/books/:_id', controller.detail);

// UPDATE
router
    .put('/books/:_id', controller.update);

// DELETE
router
    .delete('/books/:_id', controller.delete);

module.exports = router;
